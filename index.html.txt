  <script>
    const s2 = "JxHtgyrWVDJtTSJrBJRFgDUCsKI0qLljVLpqNMBm7c7Z/dRc0FjdbZAQAPsO4k7Yu2RM0BGwW0ve35yjVxrpDQ==";

    const h = "37faa4ea534fbc57dcdb9a27da4e3faa929c3e68ac5d3c0d7c5b411e34c5a859";

    let tailIndex = 81;

    function base64Encode(obj) {
      return btoa(JSON.stringify(obj));
    }

    function generateMemberLink() {
      const store = document.getElementById('store').value.trim();
      const o = document.getElementById('o').value;
      const k = document.getElementById('k').value;

      if (!store.match(/^\d+$/)) {
        alert('請輸入有效門市代碼');
        return;
      }

      const today = new Date();
      const yesterday = new Date(today);
      yesterday.setDate(today.getDate() - 1);
      const D = yesterday.toISOString().slice(0,10).replace(/-/g, '');
      const mealTime = "121500";

      const tail = String(tailIndex++).padStart(3, '0');
      const U = `TWI${store}${D}2011${tail}`;
      const S = `886KTWI${store}`;

      const qData = {
        M: "4",
        U: U,
        Country: "TW",
        S: S,
        D: D,
        T: mealTime,
        O: o,
        K: k
      };

      const qEED = base64Encode(qData);
      const link = `https://customer.kfc-listens.com/jfe/form/SV_1R1VSPEYVPlV1JA?Q_EED=${qEED}&s2=${encodeURIComponent(s2)}&cm=pki&ov=5.3.01&t=${t}&h=${h}`;

      document.getElementById('result').innerHTML = `<strong>產生連結：</strong><br><a href="${link}" target="_blank">${link}</a>`;
      document.getElementById('copyBtn').style.display = 'block';
      document.getElementById('copyBtn').setAttribute('data-link', link);
    }

    function copyToClipboard() {
      const link = document.getElementById('copyBtn').getAttribute('data-link');
      navigator.clipboard.writeText(link).then(() => {
        alert('✅ 已複製到剪貼簿！');
      });
    }
